Core handle() skeleton:
```php
public function handle(): int
{
    $dryRun = (bool) $this->option('dry-run');
    $sceneDays = (int) config('fras.retention.scene_image_days', 30);
    $faceDays = (int) config('fras.retention.face_crop_days', 90);
    $logDays = (int) config('fras.retention.access_log_retention_days', 730);
    $disk = Storage::disk('fras_events');

    $run = FrasPurgeRun::create([
        'started_at' => now(),
        'dry_run' => $dryRun,
    ]);

    [$sceneCount, $faceCount, $skipped, $logsPurged] = [0, 0, 0, 0];
    $errorSummary = null;

    try {
        // Count skipped (protected) BEFORE running purge
        $skipped = RecognitionEvent::query()
            ->whereNotNull('incident_id')
            ->whereHas('incident', fn ($i) => $i->whereNotIn('status', [IncidentStatus::Resolved, IncidentStatus::Cancelled]))
            ->where(fn ($q) => $q
                ->where(fn ($s) => $s->whereNotNull('scene_image_path')->where('captured_at', '<', now()->subDays($sceneDays)))
                ->orWhere(fn ($f) => $f->whereNotNull('face_image_path')->where('captured_at', '<', now()->subDays($faceDays)))
            )->count();

        $protectQuery = fn ($q) => $q->whereNull('incident_id')
            ->orWhereHas('incident', fn ($i) => $i->whereIn('status', [IncidentStatus::Resolved, IncidentStatus::Cancelled]));

        // Scene purge
        RecognitionEvent::query()
            ->whereNotNull('scene_image_path')
            ->where('captured_at', '<', now()->subDays($sceneDays))
            ->where($protectQuery)
            ->cursor()
            ->each(function ($event) use (&$sceneCount, $dryRun, $disk) {
                DB::transaction(function () use ($event, $dryRun, $disk) {
                    if (!$dryRun) {
                        $disk->delete($event->scene_image_path);
                        $event->update(['scene_image_path' => null]);
                    }
                });
                $sceneCount++;
            });

        // Face purge (same shape, face_image_path + $faceDays)
        // ... 

        // Access log self-purge
        if (!$dryRun) {
            $logsPurged = FrasAccessLog::where('accessed_at', '<', now()->subDays($logDays))->delete();
        }
    } catch (\Throwable $e) {
        $errorSummary = substr($e->getMessage(), 0, 1000);
        report($e);
    }

    $run->update([
        'finished_at' => now(),
        'face_crops_purged' => $faceCount,
        'scene_images_purged' => $sceneCount,
        'skipped_for_active_incident' => $skipped,
        'access_log_rows_purged' => $logsPurged,
        'error_summary' => $errorSummary,
    ]);

    return $errorSummary ? self::FAILURE : self::SUCCESS;
}
```
