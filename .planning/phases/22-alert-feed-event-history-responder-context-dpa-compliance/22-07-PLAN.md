Create PromoteIncidentModal.vue:
```vue
<script setup lang="ts">
import { useForm } from '@inertiajs/vue3';
import { computed } from 'vue';
import { promote as promoteAction } from '@/actions/App/Http/Controllers/FrasEventHistoryController';

const props = defineProps<{ event: RecognitionEventRow | null }>();
const emit = defineEmits(['close']);

const form = useForm({ priority: 'P2', reason: '' });

const canSubmit = computed(() => form.reason.length >= 8 && form.reason.length <= 500 && ['P1','P2','P3','P4'].includes(form.priority));

function submit() {
    if (!props.event) return;
    form.post(promoteAction(props.event.id).url, {
        preserveScroll: true,
        onSuccess: () => emit('close'),
    });
}
</script>
<template>
    <Dialog :open="event !== null" @update:open="$emit('close')">
        <DialogContent class="max-w-lg p-6 space-y-6">
            <DialogHeader>
                <DialogTitle>Promote to Incident</DialogTitle>
                <DialogDescription>Create an incident from this recognition event. You can't undo this, but you can reassign the incident afterward.</DialogDescription>
            </DialogHeader>
            <!-- PRIORITY micro-caps header + 4 radios -->
            <!-- REASON label + textarea + char counter -->
            <DialogFooter>
                <Button variant="outline" @click="$emit('close')">Cancel</Button>
                <Button :disabled="!canSubmit || form.processing" @click="submit">
                    {{ form.processing ? 'Promoting…' : 'Promote to Incident' }}
                </Button>
            </DialogFooter>
        </DialogContent>
    </Dialog>
</template>
```

```vue
<script setup lang="ts">
import { ref } from 'vue';
import { Head, router, usePage } from '@inertiajs/vue3';
import AppLayout from '@/layouts/AppLayout.vue';
import EventHistoryFilters from '@/components/fras/EventHistoryFilters.vue';
import EventHistoryTable from '@/components/fras/EventHistoryTable.vue';
import FrasEventDetailModal from '@/components/fras/FrasEventDetailModal.vue';
import PromoteIncidentModal from '@/components/fras/PromoteIncidentModal.vue';

defineOptions({ layout: AppLayout });

const props = defineProps<{
    events: any;          // Laravel paginator shape
    filters: { severity: string[]; camera_id: string|null; q: string; from: string|null; to: string|null };
    availableCameras: { id: string; camera_id_display: string; name: string }[];
    replayCounts: Record<string, number>;
}>();

const searching = ref(false);
const detailEvent = ref<any>(null);
const promoteEvent = ref<any>(null);

function applyFilters(nextFilters: typeof props.filters, opts: Record<string, any> = {}) {
    router.get('/fras/events', nextFilters, {
        preserveState: true, preserveScroll: true,
        onStart: () => (searching.value = true),
        onFinish: () => (searching.value = false),
        ...opts,
    });
}

function openDetail(row: any) { detailEvent.value = row; }
function openPromote(row: any) { promoteEvent.value = row; }
</script>
<template>
    <Head title="FRAS Events — IRMS" />
    <div class="p-6 lg:p-8 space-y-6">
        <header>
            <h1 class="text-2xl font-semibold">FRAS Events</h1>
            <p class="text-sm text-muted-foreground">Search recognition history. Filter by camera, severity, and date.</p>
        </header>
        <EventHistoryFilters
            :model-value="filters"
            :available-cameras="availableCameras"
            :searching="searching"
            @update:model-value="applyFilters($event, { replace: $event.q !== filters.q })"
        />
        <EventHistoryTable
            :events="events"
            :replay-counts="replayCounts"
            @open-detail="openDetail"
            @open-promote="openPromote"
        />
        <FrasEventDetailModal
            :event="detailEvent"
            :access-logs="[]"
            @close="detailEvent = null"
            @promote="openPromote(detailEvent); detailEvent = null;"
        />
        <PromoteIncidentModal
            :event="promoteEvent"
            @close="promoteEvent = null"
        />
    </div>
</template>
```
