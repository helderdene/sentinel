    4. Create `resources/js/components/incidents/EscalateToP1Button.vue`:
       ```vue
       <script setup lang="ts">
       import { computed } from 'vue';
       import { useForm } from '@inertiajs/vue3';
       import { TrendingUp } from 'lucide-vue-next';
       import { Button } from '@/components/ui/button';
       import { Tooltip, TooltipContent, TooltipTrigger } from '@/components/ui/tooltip';
       import overridePriority from '@/actions/App/Http/Controllers/IntakeStationController/overridePriority';

       interface TimelineEntry {
           event_data?: {
               source?: string;
           };
       }

       interface IncidentForEscalate {
           id: string;
           priority: string;
           timeline?: TimelineEntry[];
       }

       const props = defineProps<{ incident: IncidentForEscalate }>();

       const showButton = computed(() =>
           props.incident.timeline?.[0]?.event_data?.source === 'fras_recognition' &&
           props.incident.priority !== 'P1',
       );

       const form = useForm({
           priority: 'P1' as const,
           trigger: 'fras_escalate_button' as const,
       });

       function escalate(): void {
           form.post(overridePriority({ incident: props.incident.id }).url, {
               preserveScroll: true,
           });
       }
       </script>

       <template>
           <Tooltip v-if="showButton">
               <TooltipTrigger as-child>
                   <Button
                       variant="destructive"
                       :disabled="form.processing"
                       aria-label="Escalate incident to priority P1"
                       @click="escalate"
                   >
                       <TrendingUp class="mr-2 size-4" />
                       {{ form.processing ? 'Escalating…' : 'Escalate to P1' }}
                   </Button>
               </TooltipTrigger>
               <TooltipContent>
                   Raise priority to P1 and notify dispatcher. Required for supervisor approval.
               </TooltipContent>
           </Tooltip>
       </template>
       ```
