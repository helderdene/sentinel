       export function useFrasAlerts(

           useEcho<RecognitionAlertPayload>(

               'RecognitionAlertReceived',

                       pulseCamera(payload.camera_id, payload.severity);

                   // Info never arrives — backend factory filters (D-16). No defensive filter here.

           );

       import { computed, ref, type ComputedRef, type Ref } from 'vue';
       import type { FrasRailEvent, RecognitionAlertPayload } from '@/types/fras';

       const MAX_FRAS_FEED_SIZE = 50;

       /**
        * SSR-seeded + Echo-hydrated ring buffer for the IntakeStation FRAS rail.
        *
        * @param initialEvents — The recentFrasEvents Inertia prop (top 50 Critical+Warning events on page load).
        * @returns reactive frasEvents array + frasCount computed.
        */
       export function useFrasRail(initialEvents: FrasRailEvent[]): {
           frasEvents: Ref<FrasRailEvent[]>;
           frasCount: ComputedRef<number>;
       } {
           const frasEvents = ref<FrasRailEvent[]>([...initialEvents]);

           useEcho<RecognitionAlertPayload>(
               'fras.alerts',
               'RecognitionAlertReceived',
               (payload) => {
                   // If the event is already in the buffer (SSR seed or earlier echo),
                   // update its incident_id in place (factory dispatches RecognitionAlertReceived
                   // with incident_id populated AFTER the Incident create).
                   const existingIdx = frasEvents.value.findIndex(
                       (e) => e.event_id === payload.event_id,
                   );
                   if (existingIdx !== -1) {
                       frasEvents.value[existingIdx] = {
                           ...frasEvents.value[existingIdx],
                           incident_id: payload.incident_id,
                       };
                       return;
                   }

                   // New event: prepend, evict tail if over cap.
                   frasEvents.value.unshift({
                       event_id: payload.event_id,
                       severity: payload.severity,
                       camera_label: payload.camera_id_display,
                       personnel_name: payload.personnel_name,
                       personnel_category: payload.personnel_category,
                       confidence: payload.confidence,
                       captured_at: payload.captured_at,
                       incident_id: payload.incident_id,
                       face_image_path: null, // not in broadcast payload (Phase 22 signed-URL scope)
                       face_image_url: null,
                   });

                   if (frasEvents.value.length > MAX_FRAS_FEED_SIZE) {
                       frasEvents.value.length = MAX_FRAS_FEED_SIZE;
                   }
               },
           );

           const frasCount = computed(() => frasEvents.value.length);

           return { frasEvents, frasCount };
       }
