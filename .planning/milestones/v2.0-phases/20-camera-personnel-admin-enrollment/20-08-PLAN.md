let currentCameraFeatureCollection: { type: 'FeatureCollection'; features: any[] } = {
    type: 'FeatureCollection',
    features: [],
};

function buildCameraFeatureCollection(cameras: DispatchCamera[]) {
    return {
        type: 'FeatureCollection' as const,
        features: cameras
            .filter((c) => c.coordinates)
            .map((c) => ({
                type: 'Feature' as const,
                id: c.id,
                geometry: { type: 'Point' as const, coordinates: [c.coordinates!.lng, c.coordinates!.lat] },
                properties: {
                    id: c.id,
                    camera_id_display: c.camera_id_display,
                    name: c.name,
                    status: c.status,
                },
            })),
    };
}

function setCameraData(cameras: DispatchCamera[]): void {
    if (!map.value || !isLoaded.value) {
        pendingCameraData.value = cameras; // buffer if map not ready; existing convention
        return;
    }

    currentCameraFeatureCollection = buildCameraFeatureCollection(cameras);

    const source = map.value.getSource('cameras') as GeoJSONSource | undefined;
    if (source) {
        source.setData(currentCameraFeatureCollection);
        return;
    }

    map.value.addSource('cameras', {
        type: 'geojson',
        data: currentCameraFeatureCollection,
        promoteId: 'id',
    });

    map.value.addLayer({
        id: 'camera-halo',
        type: 'circle',
        source: 'cameras',
        paint: {
            'circle-radius': 18,
            'circle-color': CAMERA_STATUS_COLORS,
            'circle-opacity': 0.15,
            'circle-blur': 1,
        },
    }, 'incident-halo'); // beforeId per Pitfall 6

    map.value.addLayer({
        id: 'camera-body',
        type: 'symbol',
        source: 'cameras',
        layout: {
            'icon-image': ['concat', 'camera-', ['get', 'status']],
            'icon-size': 0.55,
            'icon-allow-overlap': true,
        },
    }, 'incident-halo');

    map.value.addLayer({
        id: 'camera-label',
        type: 'symbol',
        source: 'cameras',
        layout: {
            'text-field': ['get', 'camera_id_display'],
            'text-size': 9,
            'text-offset': [0, 1.6],
            'text-font': ['DIN Pro Bold', 'Arial Unicode MS Bold'],
        },
        paint: {
            'text-color': '#ffffff',
            'text-halo-color': '#000000',
            'text-halo-width': 1,
        },
    }, 'incident-halo');

    // Click popup
    map.value.on('click', 'camera-body', (e) => {
        const features = e.features ?? [];
        if (features.length === 0) return;
        const f = features[0];
        const coords = (f.geometry as any).coordinates.slice();
        const props = f.properties as Record<string, string>;

        const editUrl = `/admin/cameras/${props.id}/edit`;
        const html = `
            <div class="space-y-1 text-sm">
                <div class="font-medium">${escapeHtml(props.name ?? '')}</div>
                <div class="text-xs text-muted-foreground">${escapeHtml(props.camera_id_display ?? '')} • ${escapeHtml(props.status ?? '')}</div>
                <a href="${editUrl}" class="text-xs underline">Edit camera</a>
            </div>
        `;

        new mapboxgl.Popup({ closeButton: true })
            .setLngLat(coords)
            .setHTML(html)
            .addTo(map.value!);
    });

    map.value.on('mouseenter', 'camera-body', () => {
        map.value!.getCanvas().style.cursor = 'pointer';
    });
    map.value.on('mouseleave', 'camera-body', () => {
        map.value!.getCanvas().style.cursor = '';
    });
}

function updateCameraStatus(cameraId: string, status: 'online' | 'degraded' | 'offline'): void {
    if (!map.value) return;
    const source = map.value.getSource('cameras') as GeoJSONSource | undefined;
    if (!source) return;

    currentCameraFeatureCollection = {
        ...currentCameraFeatureCollection,
        features: currentCameraFeatureCollection.features.map((f: any) =>
            f.id === cameraId ? { ...f, properties: { ...f.properties, status } } : f
        ),
    };
    source.setData(currentCameraFeatureCollection);
}

// Helper for HTML-escaping popup content
function escapeHtml(s: string): string {
    return s.replace(/[&<>"']/g, (m) => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;' }[m] as string));
}
